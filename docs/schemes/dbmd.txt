// Юридическое лицо / владелец объектов. Иерархия владения
// (материнская компания → дочерние). is_root проставляется автоматически.
Table OwnerEntity {
  id integer [primary key]
  owner_name varchar [note: 'Название юр. лица']
  is_root boolean [note: 'Материнская компания (авто: True, если нет владельца)']
  owner_id integer [ref: > OwnerEntity.id, note: 'Следующий в иерархии владелец']
  ultimate_owner_id integer [ref: > OwnerEntity.id, note: 'Корневой владелец.']

  inn varchar [note: 'ИНН']
}


// Справочник категорий объектов по уровням
// (1: отраслевые, 2: назначения, 3: технология).
Table Category {
  id integer [primary key]
  object_level integer [note: 'Уровень объекта']
  category_name varchar [note: 'Значение категории']
  creator_id integer [note: 'Создатель']
}


Enum object_status {
  active
  in_project
  reconstruction
  stopped
}


// Объект производства. Иерархия: производство → цех → установка
// (уровни 1..3). Содержит адрес, категорию и владельца (OwnerEntity).
Table Object {
  id integer [primary key]
  object_name varchar [note: 'Полное название объекта']
  object_short_name varchar [note: 'Короткое название объекта']
  object_old_name varchar [note: 'Историческое название объекта']
  object_law_name varchar [note: 'Юридическое название объекта']

  // не используется
  object_class varchar [note: '(не использовать) Классификация объекта (завод, цех, очередь, установка)']

  capacity varchar [note: 'Мощность / объем производства']
  notes text [note: 'Дополнительная информация']

  hierarchy_level integer [note: 'Уровень в иерархии: производство -> цех -> установка']
  start_date date [note: 'Дата ввода в эксплуатацию.']
  is_reconstructed bool [note: 'Была ли реконструкция']

  status object_status [note: 'Состояние объекта']

  parent_object_id integer [ref: > Object.id, note: 'Родительский объект на уровень(или 2) выше в иерархии']
  owner_entity_id integer [ref: > OwnerEntity.id]
  category_id integer [ref: > Category.id, note: 'Разделение на категории: 1:отраслевые, 2:назначения, 3:технология']

  // Адрес
  country varchar
  region varchar
  city varchar
  street varchar
  house varchar [note: 'Адрес производства (уровень 1)']
  title varchar [note: 'Кодовое расположение установки (только уровень 3)']

  creator_id integer [note: 'Создатель']
  created_at datetime
  updated_at datetime
}


Enum auto_status {
  planned
  in_progress
  completed
  partial
}


// Связь «система на объекте»: какая AutomationSystem установлена на каком
// Object, статус и дата ввода, внедряющая компания.
Table ObjectSystem {
  id integer [primary key]
  object_id integer [ref: > Object.id]
  system_id integer [ref: > AutomationSystem.id]
  status auto_status [note: 'Статус ввода системы в эксплуатацию']
  implementation_date date [note: 'Дата ввода системы в эксплуатацию']
  implementor_id integer [ref: > Entity.id, note: 'Внедряющая компания']
}


Enum system_status {
  active
  planned
  unsupported
  decommisioned
}


// Автоматизированная система: экземпляр класса автоматизации, построенный
// на продукте вендора. Хранит техданные, модули и интерфейсы.
Table AutomationSystem {
  id integer [primary key]
  autosystem_name varchar [note: 'Полное название системы автоматизации']
  autosystem_short_name varchar [note: 'Короткое название системы автоматизации']

  release_year date [note: 'Дата выпуска системы']
  notes text [note: 'Дополнительная информация']

  system_status system_status [note: 'Статус системы.']

  system_class_id integer [ref: > AutomationClass.id, note: 'Тип системы (уровень и класс)']
  product_id integer [ref: > VendorProduct.id, note: 'Продукт на котором сделана система']
  creator_id integer [note: 'Создатель']

  technical_specs json [note: 'Технические характеристики (словарь)']
  modules json [note: 'Модули системы - изолированное ПО (список)']
  interfaces json [note: 'Интерфейсы системы (список)']
}


// Справочник классов автоматизации (уровень + класс, напр. L2 SCADA).
// Поддерживает составные классы (MES/MOM/АСУТП) с авто-подстановкой подсистем.
Table AutomationClass {
  id integer [primary key]
  level integer [note: 'Уровень автоматизации: L0 | L1 | L2 | L3 | L4']
  name_ru varchar [note: 'Русская аббревиатура (если есть)']
  system_class varchar [note: 'Класс системы']
  description text

  is_composite boolean [note: 'Составной класс (может включать подсистемы: MES, MOM, АСУТП)']
  includes_id integer [ref: > AutomationClass.id, note: 'Класс, автоматически подставляемый в подсистемы (у MOM — MES)']
}


// M2M: классы-подсистемы системы (для составных классов вроде MES/MOM/АСУТП).
Table AutomationSystem_subsystem_classes {
  id integer [primary key]
  automationsystem_id integer [ref: > AutomationSystem.id]
  automationclass_id integer [ref: > AutomationClass.id]
}


// M2M: классы-подсистемы продукта (для составных классов).
Table VendorProduct_subsystem_classes {
  id integer [primary key]
  vendorproduct_id integer [ref: > VendorProduct.id]
  automationclass_id integer [ref: > AutomationClass.id]
}


Enum product_type {
  software [note: "ПО"]
  hardware [note: "Технические средства"]
  complex [note: "ПАК"]
}


// Справочник (каталог) продуктов вендоров: ПО / технические средства / ПАК,
// на которых строятся системы автоматизации.
Table VendorProduct {
  id integer [primary key]
  product_name varchar [note: 'Название вендорского продукта']
  vendor_id integer [ref: > VendorProfile.id, note: 'Вендор (только vendor / full_cycle_vendor — через VendorProfile)']
  product_type product_type [note: 'Тип продукта']
  system_class_id integer [ref: > AutomationClass.id, note: 'Класс (ссылка на один конкретный уровень класса систем)']
  description text [note: 'Описание']
  version varchar [note: 'Версия']
  release_year date [note: 'Дата выпуска']
  end_of_support date [note: 'Конец поддержки']
  technical_specs json [note: 'Технические характеристики']
}


// M2M: отрасли применения продукта — ссылки на категории 1-го уровня.
Table VendorProduct_industries {
  id integer [primary key]
  vendorproduct_id integer [ref: > VendorProduct.id]
  category_id integer [ref: > Category.id, note: 'Отрасль — категория 1-го уровня']
}


Enum entity_type {
  vendor
  system_integrator
  engineering_company
  full_cycle_vendor
  supplier
}


// Участник рынка (вендор / системный интегратор / инжиниринговая компания /
// вендор полного цикла / поставщик). «Зонтичная» таблица: общие поля для всех
// типов, специфика типа вынесена в профили (см. ниже).
Table Entity {
  id integer [primary key]
  entity_name varchar
  inn varchar [unique, note: 'ИНН']
  website varchar [note: 'Ссылка на сайт']
  kam_name varchar [note: 'Имя КАМ коммерсант Key account manager']
  kam_phone varchar [note: 'Номер КАМ']
  contact_person varchar [note: 'Контакт от ЦК ПА']
  contact_phone varchar [note: 'Телефон от ЦК ПА']
  presentation_url varchar [note: 'Ссылка на презентацию']

  profile text [note: 'Профиль компании']
  is_partner boolean [note: 'Является ли партнером']
  registration_date date [note: 'Дата регистрации']

  entity_type entity_type [note: 'Тип участника: вендор, системный интегратор, инжиниринговая компания, вендор полного цикла, поставщик']

  financial_data json [note: 'Финансовые показатели (пока не используем)']
  contacts json [note: 'Контактная информация (список)']
}

// Профиль вендора (типы vendor / full_cycle_vendor).
// Почти пустой по полям, но служит точкой привязки продуктов:
// VendorProduct.vendor ссылается именно на VendorProfile.
Table VendorProfile {
  id integer [primary key]
  entity_id integer [ref: - Entity.id, unique, note: 'OneToOne к Entity (тип vendor / full_cycle_vendor)']
}


// Профиль поставщика (типы supplier / full_cycle_vendor).
// В отличие от вендора (авторство продукта через VendorProduct.vendor),
// поставщик связан с продуктами МНОЖЕСТВЕННОЙ связью со своей стороны:
// один поставщик поставляет много продуктов, один продукт может
// поставляться многими поставщиками (независимо от автора-вендора).
Table SupplierProfile {
  id integer [primary key]
  entity_id integer [ref: - Entity.id, unique, note: 'OneToOne к Entity (тип supplier / full_cycle_vendor)']
}


// Поставляемые продукты — множественная связь (M2M) со стороны поставщика.
Table SupplierProfile_products {
  id integer [primary key]
  supplierprofile_id integer [ref: > SupplierProfile.id]
  vendorproduct_id integer [ref: > VendorProduct.id]
}


// Профиль системного интегратора (тип system_integrator).
// managing_owner — управляющая компания из OwnerEntity (внутренний интегратор
// холдинга), необязательная. vendor_partners — вендоры-партнёры (M2M к
// VendorProfile); обратная связь даёт «карту партнёров» для вендора.
Table SystemIntegratorProfile {
  id integer [primary key]
  entity_id integer [ref: - Entity.id, unique, note: 'OneToOne к Entity (тип system_integrator)']
  managing_owner_id integer [ref: > OwnerEntity.id, note: 'Управляющая компания (необязательно; внутренний интегратор холдинга)']
}


// Вендоры-партнёры системного интегратора — множественная связь (M2M) с VendorProfile.
Table SystemIntegratorProfile_vendor_partners {
  id integer [primary key]
  systemintegratorprofile_id integer [ref: > SystemIntegratorProfile.id]
  vendorprofile_id integer [ref: > VendorProfile.id]
}

// Узкая компетенция инж. компании по функции — совокупность «класс систем +
// индустрия». Каждая строка = одна связанная пара (напр. MES · Нефтехимия);
// компания может иметь несколько таких пар.
Table SystemIntegrator_FunctionCompetency {
  id integer [primary key]
  profile_id integer [ref: > SystemIntegratorProfile.id, note: 'Профиль инж. компании']
  system_class_id integer [ref: > AutomationClass.id, note: 'Класс систем']
  industry_id integer [ref: > Category.id, note: 'Индустрия — категория 1-го уровня']
}

// Профиль инжиниринговой компании (типы engineering_company / full_cycle_vendor).
// Регион деятельности, вхожий объект и узкие компетенции (по продуктам и по функции).
Table EngineeringCompanyProfile {
  id integer [primary key]
  entity_id integer [ref: - Entity.id, unique, note: 'OneToOne к Entity (тип engineering_company)']

  region varchar [note: 'Локация — регион (значения-подсказки из Object.region, без жёсткой связи)']
  resident_object_id integer [ref: > Object.id, note: 'Вхожий объект: компания уже закрепилась и работает на этом объекте']
}


// Узкая компетенция инж. компании по продуктам — множественная связь с продуктом.
Table EngineeringCompanyProfile_product_competencies {
  id integer [primary key]
  engineeringcompanyprofile_id integer [ref: > EngineeringCompanyProfile.id]
  vendorproduct_id integer [ref: > VendorProduct.id]
}

// Узкая компетенция инж. компании по функции — совокупность «класс систем +
// индустрия». Каждая строка = одна связанная пара (напр. MES · Нефтехимия);
// компания может иметь несколько таких пар.
Table EngineeringCompany_FunctionCompetency {
  id integer [primary key]
  profile_id integer [ref: > EngineeringCompanyProfile.id, note: 'Профиль инж. компании']
  system_class_id integer [ref: > AutomationClass.id, note: 'Класс систем']
  industry_id integer [ref: > Category.id, note: 'Индустрия — категория 1-го уровня']
}

// Вендор полного цикла — dedicated профиль
Table FullCycleVendorProfile {
  id integer [primary key]
  entity_id integer [ref: - Entity.id, unique, note: 'OneToOne к Entity (тип full_cycle_vendor)']

  region varchar [note: 'Локация — регион (значения-подсказки из Object.region, без жёсткой связи)']
  resident_object_id integer [ref: > Object.id, note: 'Вхожий объект']
}

// Отдельная таблица компетенций по функции для full_cycle_vendor
Table FullCycleFunctionCompetency {
  id integer [primary key]
  profile_id integer [ref: > FullCycleVendorProfile.id, note: 'Профиль вендора полного цикла']
  system_class_id integer [ref: > AutomationClass.id]
  industry_id integer [ref: > Category.id, note: 'Индустрия — категория 1-го уровня']
}

// Поставляемые продукты (M2M)
Table FullCycleVendorProfile_products {
  id integer [primary key]
  fullcyclevendorprofile_id integer [ref: > FullCycleVendorProfile.id]
  vendorproduct_id integer [ref: > VendorProduct.id]
}
