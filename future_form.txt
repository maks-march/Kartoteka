<script>
    function toggleAdvanced() {
        const advancedSection = document.getElementById('advancedSearch');
        advancedSection.classList.toggle('show');
        
        // Опционально: менять текст кнопки
        const btn = event.target;
        if (advancedSection.classList.contains('show')) {
            btn.innerText = 'Скрыть поиск ▲';
        } else {
            btn.innerText = 'Расширенный поиск ▼';
        }
    }

    // Скрипт для того, чтобы при открытом поиске (если фильтр активен) секция не схлопывалась после перезагрузки
    window.onload = function() {
        const urlParams = new URLSearchParams(window.location.search);
        if (urlParams.has('category')) {
            document.getElementById('advancedSearch').classList.add('show');
        }
    }
</script>
